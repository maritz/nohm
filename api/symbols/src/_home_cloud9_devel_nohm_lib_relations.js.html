 10 /**
 11  * Check if the object has a relation to another object.
 12  */
 13 exports.has = function has(obj, name) {
 14   var callback = h.getCallback(arguments),
 15   self = this;
 16   name = name && typeof name !== 'function' ? name : 'child';
 17   if (!this.id || !obj.id) {
 18     Nohm.logError('Calling has() even though either the object itself or the relation does not have an id.');
 19   }
 20   this.getClient().sismember(Nohm.prefix.relations + this.modelName + ':' + name + ':' +
 21                   obj.modelName + ':' + this.id,
 22                   obj.id,
 23                   function (err, value) {
 24                     if (err) {
 25                       this.logError(err);
 26                     }
 27                     callback(err, !!value);
 28                   });
 29 };

 50                   function (err, value) {
 51                     if (err) {
 52                       self.logError(err);
 53                     }
 54                     if (!Array.isArray(value)) {
 55                       console.log('getAll has got a non-array value');
 56                       value = [];
 57                     } else {
 58                       value = value.map(function (val) {
 59                         return parseInt(val.toString(), 10);
 60                       });
 61                     }
 62                     callback(err, value);
 63                   });

 84 var allowedLinkTypes = ['sadd', 'srem'];

 86 exports.__linkProxied = function __linkProxied(type, obj, name, cb) {
 87   var self = this,
 88   parentName = name === 'child' ? 'parent' : name + 'Parent',
 89   client = self.getClient(),
 90   redisExec = function (err) {
 91     if (!err || typeof err === 'function') {
 92       var dbVals = [{
 93           key: self.modelName+':'+name+':'+obj.modelName+':'+self.id, 
 94           keyStore: self.modelName+':'+self.id,
 95           id: obj.id
 96         }, {
 97           key: obj.modelName+':'+parentName+':'+self.modelName+':'+obj.id,
 98           keyStore: obj.modelName+':'+obj.id,
 99           id: self.id
100         }
101       ];
102       async.forEach(dbVals, 
103         function (val, callback) {
104           client[type](Nohm.prefix.relationKeys+val.keyStore, Nohm.prefix.relations+val.key);
105           client[type](Nohm.prefix.relations+val.key, val.id, callback);
106         }, 
107         function (err) {
108           cb(err);
109         }
110       );
111     } else {
112       cb(err);
113     }
114   };
115   
116   if (allowedLinkTypes.indexOf(type) === -1) {
117     cb('link/unlink type invocation wrong');
118   } else if (!this.id) {
119     cb('save parent object before adding a child.');
120   } else if (!obj.id) {
121     obj.save(redisExec);
122   } else {
123     redisExec();
124   }
125 };

135 /**
136  *  Adds a reference to another object.
137  */
138 exports.link = function link(obj, name, directChange) {
139   var callback = h.getCallback(arguments);
140   name = name && typeof name !== 'function' ? name : 'child';
141   directChange = typeof directChange !== 'function' ? !!directChange : false;
142   if (this.id && directChange && obj.valid()) {
143     this.__link(obj, name, callback);
144   } else {
145     this.relationChanges.push({
146       action: 'link',
147       object: obj,
148       name: name,
149       callback: callback
150     });
151   }
152 };

154 /**
155  *  Removes the reference in the current object to
156  *  the object given in the first argument.
157  *  Note: this leaves the given object itself intact.
158  */
159 exports.unlink = function unlink(obj, name, directChange) {
160   var callback = h.getCallback(arguments),
161   change;
162   name = name && typeof name !== 'function' ? name : 'child';
163   if (this.id && directChange && obj.id) {
164     this.__unlink(obj, name, callback);
165   } else if (!this.id) {
166     /* the object hasn't been added to the db yet.
167      * this means it's still in the relationchanges array
168      * and we can just take it out there.
169      * (or it was never added, in which case nothing happens here.)
170      */
171     for (change in this.relationChanges) {
172       if (this.relationChanges.hasOwnProperty(change) &&
173           this.relationChanges[change].action === 'add' &&
174           this.relationChanges[change].name === name &&
175           checkEqual(this.relationChanges[change].object, obj)) {
176         delete this.relationChanges[change];
177       }
178     }
179   } else {
180     this.relationChanges.push({
181       action: 'unlink',
182       name: name,
183       callback: callback,
184       object: obj
185     });
186   }
187 };

189 /**
190  * Removes all links to all other object instances
191  */
192 exports.unlinkAll = function (client, callback) {
193   var self = this,
194       normalClient = this.getClient()
195       selfLinks = [];
196   client = client || normalClient;
197   
198   // this.getClient() here because client can (and should) be a multi()
199   normalClient.smembers(Nohm.prefix.relationKeys+this.modelName+':'+this.id, function (err, keys) {
200     var others = [];
201     keys.forEach(function (key) {
202       matches = key.match(/:([\w]*):([\w]*):[\d]+$/i);
203       selfName = matches[1];
204       if (matches[1] === 'child') {
205         otherName = 'parent';
206       } else if (matches[1] === 'parent') {
207         otherName = 'child';
208       } else {
209         namedMatches = matches[1].match(/^([\w]*)Parent$/);
210         if (namedMatches) {
211           selfName = matches[1]+'Parent';
212           otherName = namedMatches[1];
213         } else {
214           selfName = matches[1];
215           otherName = matches[1] + 'Parent';
216         }
217       }
218       others.push({
219         model: matches[2],
220         selfName: selfName,
221         otherName: otherName
222       });
223     });
224     async.forEach(others, function (item, cb) {
225       normalClient.smembers(
226         Nohm.prefix.relations+self.modelName+':'+item.selfName+':'+item.model+':'+self.id,
227         function (err, ids) {
228           if (err) {
229             Nohm.logError(err);
230           }
231           ids.forEach(function (id) {
232             client.srem(Nohm.prefix.relations+item.model+':'+item.otherName+':'+self.modelName+':'+id, self.id);
233           });
234           selfLinks.push(Nohm.prefix.relations+self.modelName+':'+item.selfName+':'+item.model+':'+self.id);
235           cb(err);
236         }
237       );
238     }, function (err) {
239       if (selfLinks.length > 0) {
240         client.del.apply(client, selfLinks);
241       }
242       callback(err);
243     });
244   });
245 };
